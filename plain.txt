def rolling_splits(
    df,
    feature_cols,
    target_col,
    train_window=36,
    horizon=1,
    step=1,
    dropna=True
):
    """
    Generate rolling train/test splits for time-series forecasting.
    """
    data = df.copy()
    if dropna:
        data = data.dropna(subset=feature_cols + [target_col])

    n = len(data)
    last_start = n - horizon

    for end_train in range(train_window, last_start, step):
        train = data.iloc[end_train - train_window:end_train]
        test  = data.iloc[end_train + horizon - 1:end_train + horizon]

        X_train = train[feature_cols]
        y_train = train[target_col]
        X_test = test[feature_cols]
        y_test = test[target_col]

        yield X_train, y_train, X_test, y_test
